import json
import re
import math
import google.antigravity as ad

class JSONDeepExplorer:
    def __init__(self):
        # Padrões de Segredos (Regex avançado para 2026)
        self.signatures = {
            "AWS_Key": r"AKIA[0-9A-Z]{16}",
            "JWT_Token": r"eyJ[A-Za-z0-9-_=]+\.eyJ[A-Za-z0-9-_=]+\.?[A-Za-z0-9-_.+/=]*",
            "Google_API": r"AIza[0-9A-Za-z-_]{35}",
            "Private_IP": r"10\.\d{1,3}\.\d{1,3}\.\d{1,3}|172\.(1[6-9]|2[0-9]|3[0-1])\.\d{1,3}\.\d{1,3}|192\.168\.\d{1,3}\.\d{1,3}",
            "PII_CPF": r"\d{3}\.\d{3}\.\d{3}-\d{2}", # Relevante para alvos brasileiros
            "Cloud_Metadata": r"169\.254\.169\.254"
        }
        
        # Palavras-chave para Mass Assignment e Shadow APIs
        self.risk_keys = ["admin", "role", "permission", "internal", "debug", "state", "config"]

    def _calculate_entropy(self, data):
        """Calcula a entropia de Shannon para distinguir strings aleatórias (chaves) de texto comum."""
        if not data or not isinstance(data, str): return 0
        entropy = 0
        for x in range(256):
            p_x = float(data.count(chr(x))) / len(data)
            if p_x > 0:
                entropy += - p_x * math.log2(p_x)
        return entropy

    def _analyze_value(self, key, value, path):
        """Analisa o valor individual em busca de riscos específicos."""
        findings = []
        val_str = str(value)

        # 1. Verificação de Assinaturas Conhecidas
        for name, pattern in self.signatures.items():
            if re.search(pattern, val_str):
                findings.append({
                    "type": f"CRITICAL_SIGNATURE_{name}",
                    "path": path,
                    "confidence": "HIGH"
                })

        # 2. Análise de Entropia (Segredos não estruturados)
        if len(val_str) > 12 and self._calculate_entropy(val_str) > 3.8:
            # Evita falsos positivos com URLs ou datas
            if not val_str.startswith("http") and "-" not in val_str:
                findings.append({
                    "type": "HIGH_ENTROPY_SECRET",
                    "path": path,
                    "entropy": round(self._calculate_entropy(val_str), 2)
                })

        # 3. Identificação de Candidatos a Mass Assignment
        if any(risk in key.lower() for risk in self.risk_keys):
            findings.append({
                "type": "MASS_ASSIGNMENT_CANDIDATE",
                "path": path,
                "hint": f"Tente injetar este campo em requisições POST/PUT para escalar privilégios."
            })

        return findings

    def explore(self, payload):
        """Inicia a exploração recursiva do payload JSON."""
        try:
            data = json.loads(payload) if isinstance(payload, str) else payload
            results = self._recursive_dive(data)
            
            # Automação de Categoria
            category = self._categorize_json(data)
            
            return {
                "category": category,
                "findings": results,
                "stats": {"total_keys": len(results)}
            }
        except Exception as e:
            return {"error": f"Failed to parse JSON: {str(e)}"}

    def _recursive_dive(self, data, path="root"):
        results = []
        if isinstance(data, dict):
            for k, v in data.items():
                current_path = f"{path}.{k}"
                results.extend(self._analyze_value(k, v, current_path))
                results.extend(self._recursive_dive(v, current_path))
        elif isinstance(data, list):
            for i, item in enumerate(data):
                results.extend(self._recursive_dive(item, f"{path}[{i}]"))
        return results

    def _categorize_json(self, data):
        """Identifica o propósito do arquivo para ajustar a estratégia de ataque."""
        keys = str(data.keys()) if isinstance(data, dict) else ""
        if "version" in keys and "dependencies" in keys: return "MANIFEST_FILE"
        if "host" in keys or "port" in keys or "db" in keys: return "CONFIG_FILE"
        if "data" in keys or "items" in keys: return "API_RESPONSE"
        return "GENERIC_DATA"

Para medir a similaridade entre duas versões de um esquema, a skill utiliza o Índice de Jaccard:
$$J(A, B) = \frac{|A \cap B|}{|A \cup B|}$$ Onde $A$ e $B$ são os conjuntos de chaves dos JSONs. Se $J < 1.0$, o agente Antigravity dispara um alerta de "Surface Change Detected".

Estratégias de Ataque (Red Team):
Security Regression Detection:

Cenário: Um endpoint /api/v1/login que retornava apenas um token JWT passa a retornar também o last_ip_address.

Ataque: Indica que o time de dev alterou o DTO (Data Transfer Object) e pode ter exposto campos sensíveis acidentalmente.

Zombie Endpoints Recovery:

Cenário: O recon detecta que o arquivo swagger.json.bak (detectado pela skill de dorking) contém um endpoint /v0/export que não existe no swagger.json atual.

Ataque: Tentar acessar o endpoint legado. Frequentemente, as proteções de WAF são removidas para rotas "desativadas", mas o código continua rodando no servidor.

Shadow Parameter Fuzzing:

Ataque: Como o campo é novo, há uma alta probabilidade de que ele não tenha sido validado contra SQL Injection ou NoSQLi tão rigorosamente quanto os campos antigos.

Ataque: Muitos WAFs têm regras rígidas para strings, mas falham ao analisar arrays ou objetos complexos, permitindo o bypass de filtros de assinatura.

Implementação do Motor de Drift: drift_engine.py
import google.antigravity as ad
import hashlib
import json

class SchemaDriftAnalyzer:
    def __init__(self, db_client):
        self.db = db_client # Referência ao banco de dados de snapshots do Antigravity

    def _generate_schema_hash(self, data):
        """Cria uma assinatura estrutural ignorando os valores, focando nos tipos e chaves."""
        def get_structure(obj):
            if isinstance(obj, dict):
                return {k: get_structure(v) for k, v in obj.items()}
            elif isinstance(obj, list):
                return [get_structure(obj[0])] if obj else []
            else:
                return str(type(obj))
        
        struct_str = json.dumps(get_structure(data), sort_keys=True)
        return hashlib.sha256(struct_str.encode()).hexdigest(), struct_str

    def check_for_drift(self, target_id, current_payload):
        """Compara o snapshot atual com o último salvo no diretório do agente."""
        current_hash, current_struct = self._generate_schema_hash(current_payload)
        
        # Recupera o último estado conhecido do diretório .agents/snapshots/
        last_snapshot = self.db.get_last_snapshot(target_id)
        
        if not last_snapshot:
            self.db.save_snapshot(target_id, current_hash, current_struct)
            return {"status": "NEW_TARGET", "message": "Primeiro snapshot gerado."}

        if last_snapshot['hash'] != current_hash:
            diff = self._get_diff(json.loads(last_snapshot['struct']), json.loads(current_struct))
            
            # Alerta o agente Antigravity para uma possível mudança de superfície
            ad.log_event(f"DRIFT DETECTADO em {target_id}", severity="MEDIUM")
            
            return {
                "status": "DRIFT_DETECTED",
                "delta": diff,
                "impact_score": self._calculate_impact(diff)
            }
            
        return {"status": "STABLE", "message": "Nenhuma mudança na estrutura."}

    def _get_diff(self, old, new):
        """Identifica chaves adicionadas ou removidas."""
        old_keys = set(old.keys()) if isinstance(old, dict) else set()
        new_keys = set(new.keys()) if isinstance(new, dict) else set()
        
        return {
            "added": list(new_keys - old_keys),
            "removed": list(old_keys - new_keys)
        }

    def _calculate_impact(self, diff):
        """Heurística para definir se a mudança é crítica para o Red Team."""
        critical_keywords = ['admin', 'config', 'auth', 'pass', 'token', 'key']
        impact = 0
        for key in diff['added']:
            if any(word in key.lower() for word in critical_keywords):
                impact += 50 # Alta probabilidade de nova falha
        return min(impact, 100)

BOLA (Broken Object Level Authorization) Probing:

Ação: Se um endpoint como /api/v1/order/123 é detectado, o fuzzer testa variações do ID (122, 124, UUIDs, 0, -1).

Refinamento: Ele compara o tamanho da resposta (Content-Length) e o tempo de resposta para identificar acessos bem-sucedidos a dados de outros usuários.

Verb Tampering & Method Override:

Ação: Tenta acessar um endpoint GET usando POST, PUT ou DELETE.

Estratégia: Injeta headers de override como X-HTTP-Method-Override: PUT para tentar bypassar restrições de firewall que só bloqueiam o método na linha de requisição, mas não no header.

Recursive Sub-resource Discovery:

Ação: Se encontrar /api/users, ele testa automaticamente /api/users/export, /api/users/debug, /api/users/v2.

Inteligência: Usa wordlists baseadas no setor do alvo (fintech, health, e-commerce) para prever nomes de sub-recursos prováveis.

HPP (HTTP Parameter Pollution):

Ação: Envia o mesmo parâmetro múltiplas vezes com valores diferentes: ?id=123&id=456.

Objetivo: Verificar se o backend processa o primeiro, o último ou todos, o que pode causar erros de lógica ou bypass de filtros de segurança.

import google.antigravity as ad
import requests # Ou a biblioteca de transporte nativa do seu AG

class APIEndpointFuzzer:
    def __init__(self, target_config):
        self.target = target_config
        self.probes = {
            "nosql": ['{"$gt": ""}', '{"$ne": null}'],
            "traversal": ["../", "..%2f", "%2e%2e%2f"],
            "type_juggling": [True, 0, [], {}, None]
        }

    def fuzz_parameter(self, endpoint, param_name, original_value):
        """Executa mutações específicas em um parâmetro mantendo o resto do payload estável."""
        results = []
        
        # Teste 1: Type Juggling
        for probe in self.probes["type_juggling"]:
            payload = {param_name: probe}
            results.append(self._send_probe(endpoint, "POST", payload))

        # Teste 2: BOLA/IDOR (se o valor for numérico)
        if str(original_value).isdigit():
            for offset in [-1, 1, 1000]:
                new_val = int(original_value) + offset
                results.append(self._send_probe(f"{endpoint}/{new_val}", "GET"))

        return self._analyze_results(results)

    def _send_probe(self, url, method, data=None):
        # Aqui o agente Antigravity usaria a skill 'tls-persona-mimic' para enviar a req
        # Simulando o retorno para fins de lógica:
        response = ad.transport.send(method=method, url=url, json=data)
        return {
            "method": method,
            "url": url,
            "status": response.status_code,
            "length": len(response.text),
            "time": response.elapsed_time
        }

    def _analyze_results(self, results):
        """Analisa anomalias de resposta (Diferenças de código 200 vs 403, etc)."""
        baseline = results[0] # Assume a primeira como base
        anomalies = []
        for r in results[1:]:
            if r["status"] != baseline["status"] or abs(r["length"] - baseline["length"]) > 50:
                anomalies.append(r)
        return anomalies
